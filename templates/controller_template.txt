const Plane = require("../models/planeModel");
const APIFeatures = require("../utils/apiFeatures");

exports.getAllPlanes = async (req, res) => {
  try {
    // EXECUTE QUERY
    const features = new APIFeatures(Plane.find(), req.query).filter().sort().limitFields().paginate();

    const planes = await features.query;

    //SEND RESPONSE
    res.status(200).json({
      status: "success",
      requestedAt: req.requestTime,
      results: planes.length,
      data: {
        planes,
        // Ici tours est équivalent à tours: tours
      },
    });
  } catch (err) {
    res.status(404).json({
      status: "fail",
      message: "We have not found anything !!",
    });
  }
};
